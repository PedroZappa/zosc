55std::string ZoscMessage::serialize() const {
56 std::ostringstream oss; // Output stream
57
58 // Serialize the address
59 oss << _address;
60 // Align memory with '\0' padding
61 size_t padding = ((4 - (_address.size() % 4)) % 4);
62 oss.write("\0\0\0", padding);
63
64 // Serialize the type tag string
65 oss.put(',');
66 for (const auto &arg : _arguments) {
67 if (std::holds_alternative<int32_t>(arg)) {
68 oss.put('i'); // Integer type tag
69 } else if (std::holds_alternative<float>(arg)) {
70 oss.put('f'); // Float type tag
71 } else if (std::holds_alternative<std::string>(arg)) {
72 oss.put('s'); // String type tag
73 } else if (std::holds_alternative<std::vector<uint8_t>>(arg)) {
74 oss.put('b'); // Blob type tag
75 } else if (std::holds_alternative<uint32_t>(arg)) {
76 oss.put('t'); // Time tag type
77 }
78 }
79 // Align memory with '\0' padding
80 padding = ((4 - ((_arguments.size() + 1) % 4)) % 4);
81 oss.write("\0\0\0", padding);
82
83 // Serialize the arguments
84 for (const auto &arg : _arguments) {
85 // Serialize int
86 if (std::holds_alternative<int32_t>(arg)) {
87 int32_t value = htonl(std::get<int32_t>(arg));
88 oss.write(reinterpret_cast<const char *>(&value), sizeof(value));
89 // Serialize float arg
90 } else if (std::holds_alternative<float>(arg)) {
91 float value = std::get<float>(arg);
92 uint32_t networkValue;
93 memcpy(&networkValue, &value, sizeof(networkValue));
94 networkValue = htonl(networkValue);
95 oss.write(reinterpret_cast<const char *>(&networkValue),
96 sizeof(networkValue));
97 // Serialize string arg
98 } else if (std::holds_alternative<std::string>(arg)) {
99 const std::string &value = std::get<std::string>(arg);
100 oss << value;
101 // Align memory with '\0' padding
102 padding = ((4 - (value.size() % 4)) % 4);
103 oss.write("\0\0\0", padding);
104 // Serialize blob arg
105 } else if (std::holds_alternative<std::vector<uint8_t>>(arg)) {
106 const auto &value = std::get<std::vector<uint8_t>>(arg);
107 int32_t size = htonl(static_cast<int32_t>(value.size()));
108 oss.write(reinterpret_cast<const char *>(&size), sizeof(size));
109 oss.write(reinterpret_cast<const char *>(value.data()), value.size());
110 // Align memory with '\0' padding
111 padding = ((4 - (value.size() % 4)) % 4);
112 oss.write("\0\0\0", padding);
113 }
114 }
115
116 return (oss.str());
117}

122ZoscMessage ZoscMessage::deserialize(const std::string &data) {
123 ZoscMessage message;
124 size_t pos = 0;
125
126 // Extract the address
127 size_t addressEnd = data.find('\0', pos);
128 if (addressEnd == std::string::npos)
129 throw std::runtime_error("Malformed OSC message: Missing null "
130 "terminator for address.");
131 message._address = data.substr(pos, (addressEnd - pos));
132 pos = ((addressEnd + 4) & ~3); // Align to 4 bytes
133
134 // Extract the type tag string
135 if (data[pos] != ',')
136 throw std::runtime_error("Malformed OSC message: Missing type tag "
137 "string.");
138 size_t typeTagEnd = data.find('\0', pos);
139 std::string typeTags = data.substr((pos + 1), (typeTagEnd - pos - 1));
140 pos = ((typeTagEnd + 4) & ~3); // Align to 4 bytes
141
142 // Extract the arguments
143 for (char type : typeTags) {
144 if (type == 'i') {
145 // Extract Integer
146 if (pos + sizeof(int32_t) > data.size())
147 throw std::runtime_error("Malformed OSC message: Not enough "
148 "data for integer argument.");
149 int32_t value;
150 memcpy(&value, (data.data() + pos), sizeof(value));
151 message._arguments.emplace_back(ntohl(value));
152 pos += sizeof(value);
153 } else if (type == 'f') {
154 // Extract Float
155 if (pos + sizeof(float) > data.size())
156 throw std::runtime_error("Malformed OSC message: Not enough "
157 "data for float argument.");
158 uint32_t networkValue;
159 memcpy(&networkValue, (data.data() + pos), sizeof(networkValue));
160 networkValue = ntohl(networkValue);
161 float value;
162 memcpy(&value, &networkValue, sizeof(value));
163 message._arguments.emplace_back(value);
164 pos += sizeof(networkValue);
165 } else if (type == 's') {
166 // Extract String
167 size_t stringEnd = data.find('\0', pos);
168 if (stringEnd == std::string::npos)
169 throw std::runtime_error("Malformed OSC message: Missing null "
170 "terminator for string argument.");
171 message._arguments.emplace_back(data.substr(pos, (stringEnd - pos)));
172 pos = (stringEnd + 4) & ~3; // Align to 4 bytes
173 } else if (type == 'b') {
174 // Extract Blob
175 if (pos + sizeof(uint32_t) > data.size())
176 throw std::runtime_error("Malformed OSC message: Not enough "
177 "data for blob size.");
178 int32_t size;
179 memcpy(&size, (data.data() + pos), sizeof(size));
180 size = ntohl(size);
181 pos += sizeof(size);
182
183 if (pos + size > data.size())
184 throw std::runtime_error("Malformed OSC message: Blob size "
185 "exceeds available data.");
186 message._arguments.emplace_back(std::vector<uint8_t>(
187 (data.begin() + pos), (data.begin() + pos + size)));
188 pos += size;
189 pos = ((pos + 3) & ~3); // Align to 4 bytes
190 }
191 }
192
193 return (message);
194}

203std::ostream &operator<<(std::ostream &os, const OscArg &variant) {
204 std::visit(
205 [&os](const auto &val) {
206 using T =
207 std::decay_t<decltype(val)>; // Simplify type for comparisson
208 if constexpr (std::is_same_v<T, std::vector<uint8_t>> ||
209 std::is_same_v<T, std::vector<int32_t>> ||
210 std::is_same_v<T, std::vector<float>> ||
211 std::is_same_v<T, std::vector<double>> ||
212 std::is_same_v<T, std::deque<uint8_t>> ||
213 std::is_same_v<T, std::list<uint8_t>>) {
214 // Generic container handling
215 bool first = true;
216 for (const auto &elem : val) {
217 if (!first)
218 os << ", ";
219 if constexpr (std::is_same_v<std::decay_t<decltype(elem)>,
220 uint8_t>) {
221 os << static_cast<int>(
222 elem); // Cast uint8_t to int for readable output
223 } else {
224 os << elem;
225 }
226 first = false;
227 }
228 } else
229 os << val; // Default case for other types
230 },
231 variant);
