30ZoscBundle::ZoscTimeTag ZoscBundle::ZoscTimeTag::now() {
31 auto now = std::chrono::system_clock::now(); // Get current time
32 // Get number of seconds since last UNIX epoch
33 auto secs =
34 std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch())
35 .count();
36 // Get microseconds since last second
37 auto usecs = std::chrono::duration_cast<std::chrono::nanoseconds>(
38 now.time_since_epoch())
39 .count() %
40 1'000'000'000;
41 // Convert to NTP
42 uint64_t ntpSecs = (static_cast<uint64_t>(secs) + NTP_DIFF);
43 // Get NTP fraction
44 uint64_t ntpFrac =
45 (static_cast<uint64_t>(usecs * (1LL << 32)) / 1'000'000'000);
46
47 return (ZoscTimeTag(ntpSecs << 32 | ntpFrac));
48}

107std::string ZoscBundle::serialize() const {
108 std::ostringstream oss;
109
110 // Add the OSC bundle identifier
111 oss << "#bundle";
112 size_t padding = ((4 - (7 % 4)) % 4);
113 oss.write("\0\0\0", padding);
114
115 // Serialize the TimeTag
116 uint64_t networkTimeTag =
117 htonl(static_cast<uint32_t>(_timeTag.getValue() >> 32));
118 uint32_t lowerBits =
119 htonl(static_cast<uint32_t>(_timeTag.getValue() & 0xFFFFFFFF));
120 oss.write(reinterpret_cast<const char *>(&networkTimeTag),
121 sizeof(networkTimeTag));
122 oss.write(reinterpret_cast<const char *>(&lowerBits), sizeof(lowerBits));
123
124 // Serialize elements
125 for (const auto &element : _elements) {
126 std::string serializedElement;
127 if (std::holds_alternative<ZoscMessage>(element))
128 serializedElement = std::get<ZoscMessage>(element).serialize();
129 else if (std::holds_alternative<ZoscBundle>(element))
130 serializedElement = std::get<ZoscBundle>(element).serialize();
131
132 // Write size of the element (aligned to 4 bytes)
133 uint32_t size = htonl(static_cast<uint32_t>(serializedElement.size()));
134 oss.write(reinterpret_cast<const char *>(&size), sizeof(size));
135 oss.write(serializedElement.data(), serializedElement.size());
136 }
137
138 return (oss.str());
139}

146ZoscBundle ZoscBundle::deserialize(const std::string &data) {
147 if ((data.size() < 16) || (data.substr(0, 7) != "#bundle"))
148 throw std::runtime_error("Malformed OSC bundle: Missing #bundle "
149 "identifier.");
150
151 ZoscBundle bundle;
152 size_t pos = 8; // Skip "#bundle" and align to 8 bytes
153
154 // Extract TimeTag
155 uint32_t upperBits, lowerBits;
156 memcpy(&upperBits, data.data() + pos, sizeof(upperBits));
157 memcpy(&lowerBits, data.data() + pos + sizeof(upperBits), sizeof(lowerBits));
158 uint64_t timeTag =
159 (static_cast<uint64_t>(ntohl(upperBits)) << 32) | ntohl(lowerBits);
160 bundle.setTimeTag(ZoscTimeTag(timeTag));
161 pos += 8;
162
163 // Extract elements
164 while (pos < data.size()) {
165 if ((pos + sizeof(uint32_t)) > data.size())
166 throw std::runtime_error("Malformed OSC bundle: Missing element "
167 "size.");
168 // Measure element size
169 uint32_t size;
170 memcpy(&size, (data.data() + pos), sizeof(size));
171 size = ntohl(size);
172 pos += sizeof(size);
173 if ((pos + size) > data.size())
174 throw std::runtime_error("Malformed OSC bundle: Element size "
175 "exceeds available data.");
176 // Deserialize element
177 std::string elementData = data.substr(pos, size);
178 pos += size;
179 // Align to 4 bytes
180 pos = (pos + 3) & ~3;
181
182 // Determine element type
183 if (elementData.substr(0, 7) == "#bundle") {
184 bundle.addBundle(ZoscBundle::deserialize(elementData));
185 } else {
186 bundle.addMessage(ZoscMessage::deserialize(elementData));
187 }
188 }
189
190 return (bundle);
191}
